function formatDuration(sec: number) { const m = Math.floor(sec/60); const s = Math.floor(sec%60); return `${m}:${s<10?'0':''}${s}`; }
function formatBytes(bytes?: number, decimals = 1) {
    if (!bytes) return '0 KB';
    const k = 1024;
    const dm = decimals < 0 ? 0 : decimals;
    const sizes = ['Bytes', 'KB', 'MB', 'GB', 'TB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(dm)) + ' ' + sizes[i];
}

async function generateKeyPair() {
  const keys = await window.crypto.subtle.generateKey({ name: "RSA-OAEP", modulusLength: 2048, publicExponent: new Uint8Array([1, 0, 1]), hash: "SHA-256" }, true, ["encrypt", "decrypt"]);
  const pubParams = await window.crypto.subtle.exportKey("spki", keys.publicKey);
  const privParams = await window.crypto.subtle.exportKey("pkcs8", keys.privateKey);
  const privB64 = btoa(String.fromCharCode(...new Uint8Array(privParams)));
  const pubB64 = btoa(String.fromCharCode(...new Uint8Array(pubParams)));
  return { pubB64, privB64 };
}

async function encryptMessage(payload: any, recipientPubKey: string, myPubKey: string) {
  try {
    if (!recipientPubKey) return { content: btoa(encodeURIComponent(JSON.stringify(payload))), keyRec: "", keySend: "" }; 
    const sessionKey = await window.crypto.subtle.generateKey({ name: "AES-GCM", length: 256 }, true, ["encrypt", "decrypt"]);
    const iv = window.crypto.getRandomValues(new Uint8Array(12));
    const encContent = await window.crypto.subtle.encrypt({ name: "AES-GCM", iv: iv }, sessionKey, new TextEncoder().encode(JSON.stringify(payload)));
    const contentBytes = new Uint8Array(iv.length + encContent.byteLength);
    contentBytes.set(iv);
    contentBytes.set(new Uint8Array(encContent), iv.length);
    const contentB64 = btoa(String.fromCharCode(...contentBytes));
    const wrap = async (pubB64: string) => {
        const pub = await window.crypto.subtle.importKey("spki", Uint8Array.from(atob(pubB64), c=>c.charCodeAt(0)), {name:"RSA-OAEP", hash:"SHA-256"}, false, ["encrypt"]);
        const rawKey = await window.crypto.subtle.exportKey("raw", sessionKey);
        return btoa(String.fromCharCode(...new Uint8Array(await window.crypto.subtle.encrypt({name:"RSA-OAEP"}, pub, rawKey))));
    };
    return { content: contentB64, keyRec: await wrap(recipientPubKey), keySend: await wrap(myPubKey) };
  } catch (e) { return null; }
}

async function decryptMessage(contentB64: string, keyB64: string) {
  try {
    if (!contentB64) return null;
    if (!keyB64) return JSON.parse(decodeURIComponent(atob(contentB64))); 
    const privB64 = localStorage.getItem("aperture_private_key");
    if (!privB64) return null;
    const privKey = await window.crypto.subtle.importKey("pkcs8", Uint8Array.from(atob(privB64), c=>c.charCodeAt(0)), {name:"RSA-OAEP", hash:"SHA-256"}, false, ["decrypt"]);
    const rawSessionKey = await window.crypto.subtle.decrypt({name:"RSA-OAEP"}, privKey, Uint8Array.from(atob(keyB64), c=>c.charCodeAt(0)));
    const sessionKey = await window.crypto.subtle.importKey("raw", rawSessionKey, {name:"AES-GCM"}, false, ["decrypt"]);
    const combined = Uint8Array.from(atob(contentB64), c=>c.charCodeAt(0));
    const iv = combined.slice(0, 12);
    const data = combined.slice(12);
    const dec = await window.crypto.subtle.decrypt({name:"AES-GCM", iv:iv}, sessionKey, data);
    return JSON.parse(new TextDecoder().decode(dec));
  } catch (e) { return null; }
}

function calculateAge(dob: string) {
    if (!dob) return 0;
    const birthDate = new Date(dob);
    const ageDate = new Date(Date.now() - birthDate.getTime());
    return Math.abs(ageDate.getUTCFullYear() - 1970);
}

function formatDob(dob?: string) {
    if (!dob) return "";
    const months = ['янв', 'фев', 'мар', 'апр', 'мая', 'июн', 'июл', 'авг', 'сен', 'окт', 'ноя', 'дек'];
    const d = new Date(dob);
    if (isNaN(d.getTime())) return "";
    return `${d.getDate()} ${months[d.getMonth()]} ${d.getFullYear()} (${calculateAge(dob)} лет)`;
}

function formatLastSeenImproved(d?: string) {
    if (!d) return "";
    const now = new Date();
    const dt = new Date(d);
    const diffMs = now.getTime() - dt.getTime();
    const diffMin = Math.floor(diffMs / 60000);
    if (diffMin < 3) return 'в сети';
    const isSameDay = now.toDateString() === dt.toDateString();
    if (isSameDay) return `был в ${dt.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}`;
    const yesterday = new Date(); yesterday.setDate(now.getDate() - 1);
    if (yesterday.toDateString() === dt.toDateString()) return 'был(а) вчера';
    const twoDays = new Date(); twoDays.setDate(now.getDate() - 2);
    if (twoDays.toDateString() === dt.toDateString()) return 'был(а) недавно';
    const oneWeekMs = 7 * 24 * 60 * 60 * 1000;
    if (diffMs > oneWeekMs) return 'был(а) давно';
    return `был ${dt.toLocaleDateString('ru-RU')}`;

                                                    </div>
                                                </div>
                                                <button onClick={()=>setContactNotificationsEnabled(p=>!p)} className={"w-12 h-6 rounded-full p-0.5 transition-colors " + (contactNotificationsEnabled ? 'bg-blue-600' : 'bg-white/10') }>
                                                    <div className={"w-5 h-5 bg-white rounded-full shadow transform transition-transform " + (contactNotificationsEnabled ? 'translate-x-6' : 'translate-x-0')}></div>
                                                </button>
                                            </div>

                                            {/* Favorites */}
                                            <div className="flex items-start gap-3">
                                                <div className="w-9 h-9 rounded-full bg-white/5 flex items-center justify-center text-zinc-300"><BookOpen size={16}></BookOpen></div>
                                                <div>
                                                    <div className="text-white text-sm font-medium">Избранное</div>
                                                </div>
                                            </div>

                                            {/* Note */}
                                            <div>
                                                <h3 className="text-[10px] text-zinc-500 font-bold uppercase mb-2 tracking-widest pl-1">Заметка</h3>
                                                <div className="bg-white/5 border border-white/5 p-3 rounded-2xl text-sm text-zinc-300">
                                                    {contactNote ? contactNote : <span className="text-zinc-500">Нет заметок</span>}
                                                </div>
                                                <div className="flex justify-between mt-2 text-xs text-zinc-500"><div>Заметка видна только вам</div></div>
                                            </div>
                                        </div>

                                        {/* Tabs */}
                                        <div className="mt-4">
                                            <div className="flex gap-6 px-1 border-b border-white/5 pb-2">
                                                <button className="text-sm text-purple-400 font-medium border-b-2 border-purple-500 pb-2">Публикации</button>
                                                <button className="text-sm text-zinc-400">Подарки</button>
                                                <button className="text-sm text-zinc-400">Медиа</button>
                                                <button className="text-sm text-zinc-400">Файлы</button>
                                            </div>

                                            {/* Media gallery grid */}
                                            <div className="mt-3 grid grid-cols-3 gap-2">
                                                {mediaItems.slice(0, 12).map(mi => (
                                                    <button key={mi.id} onClick={() => setLightboxSrc({ url: mi.decoded!.file!.url, type: mi.decoded!.file!.type as "image" | "video" })} className="w-full p-0">

                                                        <div className="w-full h-28 bg-black rounded overflow-hidden">
                                                            {mi.decoded!.file!.type === 'image' ? (
                                                                <img src={mi.decoded!.file!.url} className="w-full h-full object-cover" />
                                                            ) : (
                                                                <div className="w-full h-full relative bg-black">
                                                                    <video src={mi.decoded!.file!.url} className="w-full h-full object-cover" />
                                                                    <div className="absolute inset-0 flex items-center justify-center bg-black/30"><Play size={24} className="text-white" /></div>
                                                                </div>
                                                            )}
                                                        </div>
                                                    </button>
                                                ))}
                                                {mediaItems.length === 0 && <div className="text-zinc-500 text-sm col-span-3">Нет медиа в чате</div>}
                                            </div>
                                        </div>
                                    </div>
                                )
                            )}

                            {activeRoom.type==='private' && (
                                <div className="w-full mt-auto p-6 bg-[#0a0a0f] border-t border-white/5">
                                    <button onClick={toggleBlock} className={"w-full py-3.5 rounded-xl font-bold flex items-center justify-center gap-2 transition-colors " + (isBlocked ? 'bg-zinc-800 text-white hover:bg-zinc-700' : 'bg-red-500/10 text-red-500 hover:bg-red-500/20')}>
                                        {isBlocked ? "Разблокировать пользователя" : "Заблокировать пользователя"}
                                    </button>
                                </div>
                            )}
                            {activeRoom.type!=='private' && activeRoom.is_member && activeRoom.owner_id !== myProfile?.id && (
                                 <div className="w-full mt-auto p-6 bg-[#0a0a0f] border-t border-white/5">
                                    <button onClick={()=>leaveChannel(activeRoom)} className="w-full py-3.5 rounded-xl font-bold flex items-center justify-center gap-2 bg-red-500/10 text-red-500 hover:bg-red-500/20 transition-colors">
                                        <LeaveIcon size={18}></LeaveIcon> Покинуть {roomTypeNameText}
                                    </button>
                                </div>
                            )}
                        </>
                    )}
                </div>
            </motion.div>
        )}
      </AnimatePresence>

      <ImageCropperModal
        imageSrc={cropperImageSrc}
        isOpen={showImageCropper}
        onClose={() => {
          setShowImageCropper(false);
          if (cropperImageSrc) {
            URL.revokeObjectURL(cropperImageSrc);
            setCropperImageSrc("");
          }
        }}
        onCropComplete={handleCropComplete}
      />
        </div>
  );
}
